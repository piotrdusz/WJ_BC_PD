function drury(kolor)
	2c 10+, pytanie o siłe otwarcia/wejścia
		2d 12-14 nie konkretyzuje składu
		2$(kolor) podlimitowe otwarcie
end

{dwustronna przed pasem partnera}
	(1c)
		1h 5+!h, 8+
			:drury(h)
		1s 5+!s, 8+
			:drury(s)
	(1d)
		1h 5+!h, 8+
			:drury(h)
		1s 5+!s, 8+
			:drury(s)
	(1h)
		1s 5+!s, 8+
			:drury(s)
